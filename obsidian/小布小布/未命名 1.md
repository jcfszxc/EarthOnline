void TopDriver::UpdateAckermanOdometry(double f_v, double f_steer)
{
    static double x = 0, y = 0, theta = 0;
    static ros::Time last_time = ros::Time::now();
    
    ros::Time current_time = ros::Time::now();
    double dt = (current_time - last_time).toSec();
    
    if (dt > 1.0 || dt < 0) dt = 0.04;
    last_time = current_time;
    
    double omega = 0.0;
    
    // ========== 简化公式：直接用转向角计算 ==========
    if (fabs(f_v) > 0.01) {  // 有速度时
        if (fabs(f_steer) > 0.001) {
            // 简化：omega = v / R
            // 其中 R ≈ L / sin(steer)  （近似公式）
            
            double sin_steer = sin(f_steer);
            if (fabs(sin_steer) > 0.01) {  // 避免除零
                double R = wheel_base_ / sin_steer;  // 转弯半径
                omega = f_v / R;
                
                // 添加校准系数
                const double OMEGA_SCALE = 1.0 / 90.0;  // 根据测试调整！
                omega *= OMEGA_SCALE;
                
                ROS_INFO_THROTTLE(0.5, "f_v=%.3f, steer=%.0f°, R=%.2f, omega=%.3f rad/s (%.1f°/s)", 
                                 f_v, f_steer*180/M_PI, R, omega, omega*180/M_PI);
            }
        }
    }
    
    // 更新位姿
    x += f_v * cos(theta) * dt;
    y += f_v * sin(theta) * dt;
    theta += omega * dt;
    
    // 归一化
    while (theta > M_PI) theta -= 2 * M_PI;
    while (theta < -M_PI) theta += 2 * M_PI;

    // 发布TF和odom（代码同之前）
    static tf::TransformBroadcaster odom_broadcaster;
    geometry_msgs::TransformStamped odom_trans;
    odom_trans.header.stamp = current_time;
    odom_trans.header.frame_id = "odom";
    odom_trans.child_frame_id = "base_link";
    
    odom_trans.transform.translation.x = x;
    odom_trans.transform.translation.y = y;
    odom_trans.transform.translation.z = 0.0;
    odom_trans.transform.rotation = tf::createQuaternionMsgFromYaw(theta);
    
    odom_broadcaster.sendTransform(odom_trans);

    nav_msgs::Odometry odom;
    odom.header.stamp = current_time;
    odom.header.frame_id = "odom";
    odom.child_frame_id = "base_link";

    odom.pose.pose.position.x = x;
    odom.pose.pose.position.y = y;
    odom.pose.pose.orientation = tf::createQuaternionMsgFromYaw(theta);

    odom.twist.twist.linear.x = f_v;
    odom.twist.twist.angular.z = omega;

    odom_pub_.publish(odom);
}
