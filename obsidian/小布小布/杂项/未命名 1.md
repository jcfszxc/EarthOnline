    <!-- 底盘（现为平台） -->
    <link name="base_link">
        <visual>
            <origin xyz="${-platform_rear_offset + platform_length / 2} 0 ${platform_bottom + platform_thickness / 2}" rpy="0 0 0"/>

            </geometry>

        <collision>

                     izz="2.0"/>
        </inertial>
    </link>

    <link name="rear_axle">

            <origin xyz="0 0 0" rpy="${pi/2} 0 0"/>

            <material name="black"/>
        </visual>
        <collision>
            <origin xyz="0 0 0" rpy="${pi/2} 0 0"/>

        </collision>

                     izz="0.01"/>
        </inertial>

                <geometry>
                    <cylinder radius="${wheel_radius}" length="${wheel_width}"/>
                </geometry>
                <material name="black"/>

            <collision>
                <origin xyz="0 0 0" rpy="${pi/2} 0 0"/>
                <geometry>
                    <cylinder radius="${wheel_radius}" length="${wheel_width}"/>
                </geometry>
            </collision>
            <inertial>
                <mass value="2.0"/>
                <origin xyz="0 0 0" rpy="${pi/2} 0 0"/>
                <inertia ixx="0.1" ixy="0" ixz="0" 
                         iyy="0.1" iyz="0" 
                         izz="0.1"/>
            </inertial>
        </link>

        <joint name="${prefix}_wheel_joint" type="continuous">
            <parent link="rear_axle"/>
            <child link="${prefix}_wheel"/>

            <axis xyz="0 1 0"/>
        </joint>

    <xacro:rear_wheel prefix="rear_right" reflect_y="-1"/>
    <xacro:rear_wheel prefix="rear_left" reflect_y="1"/>

    <link name="front_steer_link">

                     izz="0.01"/>
        </inertial>
    </link>

        <child link="front_steer_link"/>

        <axis xyz="0 0 1"/>

    </joint>

    <!-- 前驱动轮 -->
    <link name="front_wheel">
        <visual>
            <origin xyz="0 0 0" rpy="${pi/2} 0 0"/>
            <geometry>
                <cylinder radius="${wheel_radius}" length="${wheel_width}"/>
            </geometry>
            <material name="black"/>
        </visual>
        <collision>
            <origin xyz="0 0 0" rpy="${pi/2} 0 0"/>
            <geometry>
                <cylinder radius="${wheel_radius}" length="${wheel_width}"/>
            </geometry>
        </collision>
        <inertial>
            <mass value="2.0"/>
            <origin xyz="0 0 0" rpy="${pi/2} 0 0"/>
            <inertia ixx="0.1" ixy="0" ixz="0" 
                     iyy="0.1" iyz="0" 
                     izz="0.1"/>
        </inertial>
    </link>

    <!-- 前轮动力关节 -->
    <joint name="front_wheel_joint" type="continuous">
        <parent link="front_steer_link"/>
        <child link="front_wheel"/>
        <origin xyz="0 0 0" rpy="0 0 0"/>
        <axis xyz="0 1 0"/>
    </joint>

    <!-- 为前轮添加传动装置 -->
    <transmission name="front_wheel_trans">
        <type>transmission_interface/SimpleTransmission</type>
        <joint name="front_wheel_joint">
            <hardwareInterface>hardware_interface/VelocityJointInterface</hardwareInterface>
        </joint>
        <actuator name="front_wheel_motor">
            <hardwareInterface>hardware_interface/VelocityJointInterface</hardwareInterface>
            <mechanicalReduction>1</mechanicalReduction>
        </actuator>
    </transmission>

    <!-- 为转向系统添加传动装置 -->
    <transmission name="front_steer_trans">
        <type>transmission_interface/SimpleTransmission</type>
        <joint name="front_steer_joint">
            <hardwareInterface>hardware_interface/PositionJointInterface</hardwareInterface>
        </joint>
        <actuator name="front_steer_motor">
            <hardwareInterface>hardware_interface/PositionJointInterface</hardwareInterface>
            <mechanicalReduction>1</mechanicalReduction>
        </actuator>
    </transmission>

    <!-- 激光雷达 -->

        <collision>

            </geometry>
        </collision>
        <inertial>
            <mass value="0.5"/>
            <inertia ixx="0.01" ixy="0" ixz="0" 
                     iyy="0.01" iyz="0" 
                     izz="0.01"/>
        </inertial>
    </link>

    <joint name="lidar_joint" type="fixed">
        <parent link="base_link"/>
        <child link="lidar_link"/>
        <origin xyz="${lidar_x} 0 ${lidar_height}" rpy="0 0.174533 0"/>
    </joint>

    <!-- IMU 链接 -->
    <link name="imu_link">
        <visual>
            <geometry>
                <box size="0.05 0.05 0.02"/> <!-- IMU 小型传感器，尺寸示例 -->
